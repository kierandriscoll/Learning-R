3 Working with Dataframes or tibbles

3.1 dplyr

dplyr is a package that makes manipulating dataframes/tibbles simpler. It includes functions to subset and merge dataframes.

library(dplyr)

# Keep/Drop specific columns with select()
select(mydf,extraStat) # Only keeps extraStat
select(mydf,-extraStat) # Drop extraStat by prefixing with  a '-' 

# Keep columns that meet specific conditions select_if()
select_if(mydf,is.numeric)

# Select rows that meet specific conditions using filter()
filter(mydf,extraStat > 25000 & extraStat < 35000)

# Rename columns using rename() 
rename(mydf, MeanIncome = extraStat)  #Nb. the new name is specifed first New=Old

# Create new columns using mutate() or transmute()
mutate(mydf, newcol=extraStat*2)  
# OR to also drop all other columns 
transmute(mydf, newcol=extraStat*2)  

# Create basic statistics (eg. sum, mean, median, in, max)
summarise(mydf, sum(extraStat), mean(extraStat), max(extraStat), freq=n())
# You can name your stat by adding eg. statname = sum(extraStat)

# Sort data (ascending by default)
arrange(empdf,EmployerCount) # add desc(EmployerCount) for descending

# Group By a column - On its own this doesnt produce anything, but when combined with other functions the results will be grouped
group_by(peopledf,Name) %>%
  summarise(sum(Age))
  
# To stop R grouping you need to run:
ungroup()

# Add/Append dataframes together - they should have same columns names/types 
bind_rows(mydf1, mydf2)

# Merge dataframes by common id (other joins include inner_join, right_join)
left_join(mydf1, mydf2, by = 'id')

# Anti-Join : Only keep rows from mydf1 if they DONT match with mydf2, eg. to remove certain types of observation
anti_join(mydf1, mydf2, by = 'id')

# Extract a column as a vector/list
pull(peopletdf, Age)

# Frequency count of variable (+ optional sorting of results)
count(peopletdf, Gender, sort = TRUE)

3.2 tidyr & tidy data

tidyr includes some extra functions for converting data to/from a tidy format. Nb. tidyr includes dplyr.

library(tidyr)
## Warning: package 'tidyr' was built under R version 3.4.4
tidydf <- tibble('Year'=c(2016,2016,2017,2017,2018,2018),
                 'Type'=c('A','B','A','C','B','C'),
                 'Amount'=c(111,222,333,444,555,666) )

# Pivot tidy data so that values become columns
nottidy <- spread(tidydf, key=Year, value=Amount)

# Pivot data into a tidy format (ie. reverse spread)
gather(nottidy, key=Age, value=Amount, -Type)
## # A tibble: 9 x 3
##   Type  Age   Amount
##   <chr> <chr>  <dbl>
## 1 A     2016     111
## 2 B     2016     222
## 3 C     2016      NA
## 4 A     2017     333
## 5 B     2017      NA
## 6 C     2017     444
## 7 A     2018      NA
## 8 B     2018     555
## 9 C     2018     666
